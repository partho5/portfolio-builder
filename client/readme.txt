│   ├── layout/          # Layout components

│   ├── icons/           # Custom icons

├── types/               # TypeScript type definitions

├── services/            # API services
├── stores/              # State management
├── styles/              # Global styles
├── assets/              # Static assets
│   ├── images/

│   └── icons/

├── providers/           # Context providers

│   └── e2e/
